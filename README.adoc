== NgFx

Reactive hooks for Angular.

[source,typescript]
----
@Component({
    selector: "app-root",
    template: `
        <div>Count: {{ count }}</div>
    `,
})
export class AppComponent extends Connectable {
    @Input()
    count: number = 0

    ngOnConnect() {
        effect(() =>
            timer(1000).subscribe(() =>
                this.count += 1
            )
        )
    }
}
----

link:https://www.npmjs.com/package/ng-effects[Install via NPM]
